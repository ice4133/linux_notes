首先烧录

```bash
# 这个只是针对stm32mp157d-atk的，其他的芯片可能不一样，具体看官方文档
setenv bootargs 'console=ttySTM0,115200 root=/dev/mmcblk1p3 rootwait rw'

ext4load mmc 1:2 0xc2000000 uImage
ext4load mmc 1:2 0xc4000000 stm32mp157d-atk.dtb
bootm 0xc2000000 - 0xc4000000





# 为了永久生效，按下restart
# 第一步：设置内核参数（大脑的指令）
setenv bootargs 'console=ttySTM0,115200 root=/dev/mmcblk1p3 rootwait rw'

# 第二步：设置自动启动流程（早起闹钟的动作）
setenv bootcmd "ext4load mmc 1:2 0xc2000000 uImage; ext4load mmc 1:2 0xc4000000 stm32mp157d-atk.dtb; bootm 0xc2000000 - 0xc4000000"

# 第三步：将上述设置从 RAM 写入 eMMC 存储（点击保存）
saveenv
```
