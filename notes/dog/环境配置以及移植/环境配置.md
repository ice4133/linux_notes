# 配一些其余库
缺少 tf_transformations库，安装方法如下：
sudo apt update
sudo apt install ros-humble-tf-transformations
//还需要安装它的底层依赖
pip3 install transforms3d



缺少keyboard
pip3 install keyboard
python3.8 -m pip install keyboard
但是要下载相对应的3.8版的pip
更新包列表
sudo apt update

安装 Python 3.8 专用的 distutils 和 pip
sudo apt install python3.8-distutils python3-pip

sudo python3.8 -m pip install keyboard

python3.8 -m pip install --user --upgrade --force-reinstall numpy
报错不用管，是故意让它报错的


import serial出错

输入pip3 install pyserial
只是在调用的时候叫serial，实际上下载的是pyserial这个库，安装完成后就可以正常导入了

# 解决宇数内置sdk问题
```bash

vim ~/.bashrc


# for unittree sdk environment
export PYTHONPATH=$PYTHONPATH:~/my_dog/dog_ws/src/robot/resource/robot/sdk_lib
export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:~/my_dog/dog_ws/src/robot/resource/robot/sdk_lib
# end


# 根据自己的路径来，我自己的是my_dog/dog_ws/src/robot/resource/robot/sdk_lib

# 不同电脑肯定不一样，所以要根据自己的路径来修改上面的命令

source ~/.bashrc
```
