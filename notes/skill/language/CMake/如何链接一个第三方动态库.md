# 现代CMake写法
find_package(OpenCV REQUIRED)
target_link_libraries(dog ${OpenCV_LIBRARIES}) # 一句话自动搞定头文件和库
# 传统CMake写法
target_include_directories(dog PUBLIC ${CMAKE_CURRENT_SOURCE_DIR}/third_party/unittree/include)
// 这个是找到第三方库的头文件路径，告诉编译器去哪里找头文件了


target_link_libraries(dog ${CMAKE_CURRENT_SOURCE_DIR}/third_party/unittree/lib/libUnitreeMotorSDK_M80106_Linux64.so)
//这一句话是告诉链接器去哪里找库文件了，链接器会去这个路径下找libUnitreeMotorSDK_M80106_Linux64.so这个库文件了，要不然编译器就会报错找不到这个库了


set_target_properties(dog PROPERTIES 
  INSTALL_RPATH "${CMAKE_CURRENT_SOURCE_DIR}/third_party/unittree/lib"
  BUILD_WITH_INSTALL_RPATH TRUE
)
//这个是针对动态库的，因为系统运行的时候要找这个动态库，所以要告诉系统去哪里找这个动态库了，要不然运行的时候就会报错找不到这个动态库了



但如果把所有的库都放在系统路径下了，就不需要这些麻烦的配置了，直接写一句 target_link_libraries(dog UnitreeMotorSDK_M80106_Linux64) 就好了，系统会自动去系统路径下找这个库了。

但因为这个代码是闭源的，所以只能用链接库了
